total = 0
item_count = 0

print("========== Grocery Billing System ==========")
print("Enter the price of each item")
print("Enter 0 when billing is completed\n")

while True:
    price = float(input("Enter item price: "))

    if price == 0:
        break

    if price < 0:
        print("Invalid price. Try again.\n")
        continue

    item_count = item_count + 1
    total = total + price

print("\n========== Bill Summary ==========")
print("Items added successfully")
print("Total items purchased:", item_count)
print("Total amount to pay:", total)
print("Thank you for shopping with us")
